        self.get_response = get_response
        cred = credentials.Certificate(settings.FIREBASE_CONFIG)

            firebase_admin.get_app()

    def __call__(self, request):

        if request.user.is_authenticated:
            return self.get_response(request)

        if not (request.META.get("HTTP_AUTHORIZATION")):
            return JsonResponse(
                {
                    "message": "Debe autenticarse para realizar esta acci贸n",
                    "error": "unauthenticated"
                },
                status=PermissionDenied.status_code,
            )

        authorization = request.META.get("HTTP_AUTHORIZATION").split(" ")
        if len(authorization) != 2:
            return JsonResponse(
                {
                    "message": "Debe autenticarse para realizar esta acci贸n",
                    "error": "unauthenticated"
                },
                status=PermissionDenied.status_code,
            )
        if authorization[0] != "JWT":
            return JsonResponse(
                {
                    "message": "Debe autenticarse para realizar esta acci贸n",
                    "error": "unauthenticated"
                },
                status=PermissionDenied.status_code,
            )
        token = authorization[1]
        try:
            userinfo = auth.verify_id_token(token)
            usuario = Usuario.objects.filter(firebase_uid=userinfo["uid"])
            if usuario.count() == 0:
                user = User.objects.create(
                    first_name=userinfo["name"],
                    email=userinfo["email"],
                    username=str(int(time.time()))
                    + userinfo["name"].lower().replace(" ", ""),
                )
                user.set_password(user.username)
                user.save()
                usuario = Usuario.objects.create(
                    user=user, estado="I", firebase_uid=userinfo["uid"]
                )
            else:
                usuario = usuario[0]
                if usuario.estado == "I":
                    return JsonResponse(
                        {
                            "message": "Su usuario a煤n no fue activado, debe esperar la confirmaci贸n del administrador",
                            "error": "unauthenticated"
                        },
                        status=PermissionDenied.status_code,
                    )
            user = usuario.user
            request._force_auth_user = user
        except ExpiredIdTokenError:
            return JsonResponse(
                {
                    "message": "Su sesi贸n ha expirado",
                    "error": "unauthenticated"
                },
                status=AuthenticationFailed.status_code,
            )
        except InvalidIdTokenError:
            return JsonResponse(
                {
                    "message": "Debe autenticarse para realizar esta acci贸n",
                    "error": "unauthenticated"
                },
                status=NotAuthenticated.status_code,
            )
        return self.get_response(request)
